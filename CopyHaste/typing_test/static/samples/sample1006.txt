
def find_bins(snrsq_list):
  remaining = 0
  huge_snrs = 0
  upper_lim = 1e3
  total_count = 0
  bins		= [1e-40 for i in range(int(upper_lim))]
  biggest	 = 0
  for snrsq in snrsq_list:
	  snrsq = int(snrsq*snrsq)
	  total_count += 1
	  if snrsq > 200:
		  remaining += 1
	  if snrsq > upper_lim:
		  huge_snrs += 1
	  else:
		  if snrsq > biggest:
			  biggest = snrsq
		  bins[snrsq] += 1
  high_end = min(biggest, 200)
  return bins, biggest, high_end, remaining
