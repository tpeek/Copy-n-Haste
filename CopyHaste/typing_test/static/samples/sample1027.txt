class CRManager:
	def __init__(self, num_nodes, start_index = 0):
		self.num_nodes = num_nodes
		self.nodes = {}
		for i in xrange(num_nodes):
			self.nodes[start_index] = None
			start_index += 1
	def start_nodes(self):
		status = False
		keys = self.nodes.keys()
		for index in keys:
			status = self.start(index)
			if status == False:
				break
		return status
