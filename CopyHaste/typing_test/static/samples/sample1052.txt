def GetIndParam(means):
	vals = []
	if isinstance(means, float) or isinstance(means, int):
		std = means/10.0
		vals = np.random.normal(means, std)
		if vals < 0.00001:
			vals = 0.00001
	else:
		for val in means:
			std = val/10.0
			i = np.random.normal(val, std)
			if i < 0.00001:
				i = 0.00001
			vals.append(i)
	return vals
def GetRAD(vector):
	RAD = []
	unique = list(set(vector))
	for val in unique:
		RAD.append(vector.count(val))
	return RAD, unique
