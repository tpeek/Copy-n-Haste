def round_value(value, round_decimals_to = 3):
	number = float(10**round_decimals_to)
	return int(value * number) / number

def initialize_deck():
	deck = []
	for rank in RANKS:
		for suit in SUITS:
			deck.append(rank+suit)
	return deck
