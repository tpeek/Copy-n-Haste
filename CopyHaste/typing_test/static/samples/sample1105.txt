	options = 3
	probs = [0] * options
	for value in array:
		probs[int(value)] += 1
	for index, value in enumerate(probs):
		probs[index] = value/float(len(array))
		if probs[index] == 0.0:
			probs[index] = 0.000000000001
	total = sum(probs)
	for index, value in enumerate(probs):
		probs[index] = value/total
	return probs
