
def find_packages():
	packages = ['mdtraj.scripts']
	for dir,subdirs,files in os.walk('MDTraj'):
		package = dir.replace(os.path.sep, '.')
		if '__init__.py' not in files:
			continue
		packages.append(package.replace('MDTraj', 'mdtraj'))
	return packages
