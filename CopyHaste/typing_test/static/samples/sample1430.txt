def get_ns3_relative_path(path):
	l = []
	head = path
	while head:
		new_head, tail = os.path.split(head)
		if new_head == head:
			raise ValueError
		head = new_head
		if tail == 'ns3':
			return os.path.join(*l)
		l.insert(0, tail)
	raise AssertionError("is the path %r inside ns3?!" % path)
