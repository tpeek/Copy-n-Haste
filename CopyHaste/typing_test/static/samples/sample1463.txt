def cleanSave(content,directory,mode,**kwargs):
	try:
		os.remove(directory)
	except OSError:
		pass
	try:
		if mode == 'fig':
			content.savefig(directory,**kwargs)
		elif mode == 'anim':
			content.save(directory,**kwargs)
		content.close()
	except Exception as e:
		print "Figure",directory,'save failed, error:', e
		directory = 'null'
	del content
	return directory
