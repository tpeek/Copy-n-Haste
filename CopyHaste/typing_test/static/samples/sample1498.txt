
def description_of(lines, name='stdin'):
	u = UniversalDetector()
	for line in lines:
		u.feed(line)
	u.close()
	result = u.result
	if result['encoding']:
		return '{0}: {1} with confidence {2}'.format(name, result['encoding'],
													 result['confidence'])
	else:
		return '{0}: no result'.format(name)
