def test_repeated_save_with_alpha():
	fig = Figure([1, 0.4])
	canvas = FigureCanvas(fig)
	fig.set_facecolor((0, 1, 0.4))
	fig.patch.set_alpha(0.25)
	buf = io.BytesIO()
	fig.savefig(buf,
				facecolor=fig.get_facecolor(),
				edgecolor='none')
	buf.seek(0)
	fig.savefig(buf,
				facecolor=fig.get_facecolor(),
				edgecolor='none')
	buf.seek(0)
	assert_array_almost_equal(tuple(imread(buf)[0, 0]),
							  (0.0, 1.0, 0.4, 0.250),
							  decimal=3)
