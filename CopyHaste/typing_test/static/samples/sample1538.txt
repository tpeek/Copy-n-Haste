def exponential(x, y):
	x = x*9
	y = y*9
	x1 = x+1.0
	x2 = x-2.0
	x4 = x-4.0
	x7 = x-7.0
	y1 = x+1.0
	y2 = y-2.0
	y3 = y-3.0
	y7 = y-7.0
	f = (0.75 * np.exp(-(x2*x2+y2*y2)/4.0) +
		 0.75 * np.exp(-x1*x1/49.0 - y1/10.0) +
		 0.5 * np.exp(-(x7*x7 + y3*y3)/4.0) -
		 0.2 * np.exp(-x4*x4 -y7*y7))
	return f
exponential.title = 'Exponential and Some Gaussians'
