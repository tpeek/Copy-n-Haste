def bed_read_in(bedfile):
		out={}
		fin=open(bedfile)
		for line in fin:
				pin=line.strip().split()
				if not pin[0] in out.keys():
						out[pin[0]]=[]
				out[pin[0]].append([int(pin[1]),int(pin[2])])
		fin.close()
		return out
