def _gevent_sendfile(fdout, fdin, offset, nbytes):
	while True:
		try:
			return o_sendfile(fdout, fdin, offset, nbytes)
		except OSError as e:
			if e.args[0] == errno.EAGAIN:
				wait_write(fdout)
			else:
				raise
def patch_sendfile():
	if o_sendfile is not None:
		setattr(wsgi, "sendfile", _gevent_sendfile)
