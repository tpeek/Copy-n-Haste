
def copytree(src,dst):
	names = os.listdir(src)
	if(os.path.isdir(dst) != True):
		os.makedirs(dst)
	for name in names:
		srcname = os.path.join(src, name)
		dstname = os.path.join(dst, name)
		if os.path.isdir(srcname):
				copytree(srcname, dstname)
		else:
			shutil.copy2(srcname, dstname)
