
class TestNetflix (TestCase) :
	def test_preload(self):
		user_file, movie_file = ["1 1", "2 2", "3 3"], ["1 1", "2 2", "1 1"]
		users_avg, movies_avg, overall_mean, overall_users_mean = preload(user_file, movie_file)
		self.assertEqual(len(preload(user_file, movie_file)), 4)
	def test_preload_2(self):
		user_file, movie_file = ["1 1", "2 2", "3 3"], ["1 1", "2 2", "1 1"]
		users_avg, movies_avg, overall_mean, overall_users_mean = preload(user_file, movie_file)
		self.assertEqual((users_avg, movies_avg, overall_mean, overall_users_mean), ({1: 1.0, 2: 2.0, 3: 3.0}, {1: 1.0, 2: 2.0}, 0.00022598870056497175, 1.2495080062225499e-05))
