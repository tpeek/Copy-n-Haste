
def clear_off():
	for fn in ('LiSEworld.db', 'LiSEcode.db'):
		try:
			remove(fn)
		except OSError:
			pass
def mkengine(w='LiSEworld.db', caching=True, seed=None):
	return Engine(
		worlddb=w,
		codedb='LiSEcode.db',
		caching=caching,
		random_seed=seed
	)
