def dodir(path):
	try:
		os.makedirs(path, 0o755)
	except OSError:
		if not os.path.isdir(path):
			raise
		os.chmod(path, 0o755)

	os.chmod(dst, 0o644)

	os.spawnlp(os.P_WAIT, "bash", "bash", "-c", cmd)
