class Some:
  "Keep some things."
  def __init__(i, keep=8):
	i.n, i.any, i.keep, i.ordered = 0,[],keep, False
  def __iadd__(i,x):
	i.ordered = False
	i.n += 1
	now = len(i.any)
	if now < i.keep:
	  i.any += [x]
	elif r() <= now/i.n:
	  i.any[ int(r() * now) ]= x
	return i
  def sorted(i):
	i.any = i.any if i.ordered else sorted(i.any)
	i.ordered = True
	return i.any
  def lo(i): return i.sorted()[0]
  def hi(i): return i.sorted()[-1
