		self.size = sizein
		self.last = 0

	def __len__(self):
		return self.size
	def update_bar(self, length):
		length = int(length)
		print("[" + self.spinny[0] + "] ", end = "")
		tmp = self.spinny.popleft()
		self.spinny.append(tmp)
		if self.last > length:
			print("", end = "\n")
			raise ValueError('Your new bar value decreased!')
		self.last = length

			print("", end = "\n")
			raise ValueError('You entered a len > size for the bar')
