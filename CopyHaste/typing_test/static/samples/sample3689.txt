def getset_item(obj, item, set=None):
	for func in [lambda x: x, int]:
		try:
			item = func(item)
			if set is not None:
				obj[item] = set
			return obj[item]
		except (TypeError, LookupError, ValueError):
			pass
	raise LookupError('%s not in config' % item)
