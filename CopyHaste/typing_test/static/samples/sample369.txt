
def distance_between_points(p1, p2):
	dx = p1.x - p2.x
	dy = p1.y - p2.y
	dist = math.sqrt(dx**2 + dy**2)
	return dist
def move_rectangle(rect, dx, dy):
	rect.corner.x += dx
	rect.corner.y += dy
def move_rectangle_copy(rect, dx, dy):
	new = copy.deepcopy(rect)
	move_rectangle(new, dx, dy)
	return new
