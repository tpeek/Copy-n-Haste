def setup_logging_level():
	level = logging.DEBUG
	if "--verbose" in sys.argv or "-v" in sys.argv:
		level = flexget.logger.TRACE
	elif "--quiet" in sys.argv or "-q" in sys.argv:
		level = logging.INFO
	logging.getLogger().setLevel(level)
	return level
