def strip_html_content_to_text(html_content):
	text_content = re.sub(r"(\s|&nbsp;|//)+", " ", html_to_text(html_content))
	text_content = re.sub(r"<!\[CDATA\[.*\]\]>", "", text_content)
	text_content = re.sub(r"<!--.*-->", "", text_content)
	return text_content
