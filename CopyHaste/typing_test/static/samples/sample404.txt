def get_vpc_info(vpc):
	return({
		'id': vpc.id,
		'cidr_block': vpc.cidr_block,
		'dhcp_options_id': vpc.dhcp_options_id,
		'region': vpc.region.name,
		'state': vpc.state,
	})
