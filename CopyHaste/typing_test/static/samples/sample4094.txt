	def reset(self):
		CharSetProber.reset(self)
		for codingSM in self._mCodingSM:
			if not codingSM:
				continue
			codingSM.active = True
			codingSM.reset()
		self._mActiveSM = len(self._mCodingSM)
		self._mDetectedCharset = None
	def get_charset_name(self):
		return self._mDetectedCharset

			return 0.99
