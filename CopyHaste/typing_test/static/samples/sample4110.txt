def guess_geom_type(csv_path, geom_name):
	with open(csv_path) as file:
		for row in DictReader(file):
			geom = ogr.CreateGeometryFromWkt(row.get(geom_name))
			return geometry_types.get(geom.GetGeometryType(), False)
	
	return False
