def cached(timeout=10 * 60, key='view/%s'):
	def decorator(f):
		@wraps(f)
		def decorated_function(*args, **kwargs):
			cache_key = key % request.path
			rv = CACHE.get(cache_key)
			if rv is not None:
				return rv
			rv = f(*args, **kwargs)
			CACHE.set(cache_key, rv, timeout=timeout)
			return rv
		return decorated_function
	return decorator
