	@staticmethod
	def format_seconds(seconds):
		(mins, secs) = divmod(seconds, 60)
		(hours, mins) = divmod(mins, 60)
		if hours > 99:
			return '--:--:--'
		if hours == 0:
			return '%02d:%02d' % (mins, secs)
		else:
			return '%02d:%02d:%02d' % (hours, mins, secs)
