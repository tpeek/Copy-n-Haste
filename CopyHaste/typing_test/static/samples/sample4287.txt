class PrintSimple():
	def __init__(self, dups):
		self.dups = dups
		self.num = {0:1}
	def number(self, level):
		if self.dups:
			to_return = "%d." % level
		else:
			to_return = str(level)
			if level > 1:
				to_return += "-" + str(self.num[level-1])
			to_return += "."
			self.num[level] = 1
