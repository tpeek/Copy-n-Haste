class TestNetflix (TestCase) :
	def test_load_cache_1(self):
		cache = load_cache("thisisnotafile.txt")
		self.assertFalse(cache)
	def test_load_cache_2(self):
		cache = load_cache("jl47232-probe_actual_ratings_cache.json")
		self.assertTrue(cache)
		self.assertEqual(cache["1"]["30878"], "4")
	def test_load_cache_3(self):
		cache = load_cache("jl47232-movie_avg_cache.json")
		self.assertTrue(cache)
		self.assertEqual(cache["1"], 3.749542961608775)
	def test_netflix_predict_1(self):
		rating = netflix_predict(1,1,{1:1},{1:1})
		self.assertEqual(rating, 1)
	def test_netflix_predict_2(self):
		rating = netflix_predict(1,1,{1:4},{1:4})
		self.assertEqual(rating, 4.3)
