def get_treasury_date(dstring):
	return date_conversion(dstring.split("T")[0], date_pattern='%Y-%m-%d',
						   to_utc=False)
def get_treasury_rate(string_val):
	val = guarded_conversion(float, string_val)
	if val is not None:
		val = round(val / 100.0, 4)
	return val
