
def reddening_correction_sfd98(extinction_r):
	E_BV = extinction_r / 2.751
	A_u = E_BV * 5.155
	A_g = E_BV * 3.793
	A_r = E_BV * 2.751
	A_i = E_BV * 2.086
	A_z = E_BV * 1.479
	return (A_u, A_g, A_r, A_i, A_z)
