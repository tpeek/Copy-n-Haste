  xattrs = list()
  for k, v in elem.attributes().items():
	if k == 'label':
	  k = 'name'
	if v == None:
	  continue

	elif isinstance(v, types.FloatType):
	  xattr = E.att({'name': k, 'value': str(v), 'type': 'real'})
