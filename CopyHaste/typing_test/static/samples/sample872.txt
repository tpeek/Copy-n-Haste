
class BinFile(object):
	def set_u32(self, addr, value):
		self.a[addr+0] = (value >> 0) & 0xff
		self.a[addr+1] = (value >> 8) & 0xff
		self.a[addr+2] = (value >> 16) & 0xff
		self.a[addr+3] = (value >> 24) & 0xff
	def u32(self):
		addr = self.addr
		rv = self.a[addr]
		rv |= self.a[addr+1] << 8
		rv |= self.a[addr+2] << 16
		rv |= self.a[addr+3] << 24
		self.addr += 4
		return rv
	def u16(self):
		addr = self.addr
		rv = self.a[addr]
		rv |= self.a[addr+1] << 8
		self.addr += 2
		return rv
	def u8(self):
		rv = self.a[self.addr]
		self.addr += 1
		return rv
